        srv.controllerConnState == ConnStateDegraded ||
        srv.controllerConnState == ConnStateUnreachable ||
        srv.consecutiveHeartbeatFail >= 3

    if shouldRediscover {

                log.Printf("node-agent: failed to persist rediscovered controller endpoint: %v", err)

            log.Printf("node-agent: controller endpoint refreshed: %s -> %s", old, ep)
            loggedMissingEndpoint = false

3. Lease heartbeat ignores whether the LWT was actually applied

File: workflow_server/executor_lease.go
Function: heartbeatLoop()

Observed pattern

WHERE run_id = ?

But code uses .Exec() and checks only transport/query error, not whether the conditional update applied.

Failure mode

heartbeat loop continues happily
owner thinks it still owns the run

Why it matches today’s bug family
This is a distributed classic:

compare-and-set attempted, but applied-result ignored

Minimal safe fix
Use ScanCAS(...) and stop heartbeating if applied == false.

Patch shape

applied, err := m.srv.session.Query(`
    UPDATE workflow.executor_leases SET heartbeat_at = ?
    WHERE run_id = ?
    IF executor_id = ?`,
    now, runID, m.executorID,
).WithContext(ctx).ScanCAS(nil)

if err != nil {
    slog.Warn("executor lease: heartbeat failed", "run_id", runID, "err", err)
    continue
}
if !applied {
    slog.Warn("executor lease: ownership lost, stopping heartbeat",
        "run_id", runID, "executor_id", m.executorID)
    m.mu.Lock()
    if cancel, ok := m.ownedRuns[runID]; ok {
        cancel()
        delete(m.ownedRuns, runID)
    }
    m.mu.Unlock()
    return
}

Extra hardening
Emit a metric and structured event:

workflow.executor_lease_lost

This one is high priority.
